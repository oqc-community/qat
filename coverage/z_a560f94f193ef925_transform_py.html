15class DesugaringPass(TransformPass): 

16 """Transforms syntactic sugars and implicit effects. 

17 

18 The goal of this pass is to desugar any syntactic and semantic sugars. One of these 

19 sugars is iteration constructs such as :class:`Sweep` and :class:`Repeat`. 

20 

21 1- Adds an iterator variable for sweep loops. We state that sweep loops implicitly 

22 declare an iterator variable to control progress of the sweep. See :class:`BindingPass` 

23 """ 

24 

25 def run(self, ir: InstructionBuilder, res_mgr: ResultManager, *args, **kwargs): 

26 """:param ir: The list of instructions stored in an :class:`InstructionBuilder`.""" 

27 

28 for inst in ir.instructions: 

29 if isinstance(inst, Sweep): 

30 count = len(next(iter(inst.variables.values()))) 

31 iter_name = f"{hash(inst)}" 

32 inst.variables[iter_name] = np.linspace(1, count, count, dtype=int).tolist() 

33 return ir
