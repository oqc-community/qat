3"""Attributes for the ``q1_scf`` structured control flow dialect. 

4 

5The dialect reuses the ``q1_cf`` predicate enumerations for its conditions (see 

6:mod:`qat.experimental.dialect.q1_cf.ir.attrs`) and defines one attribute of its 

7own: 

8 

9- :class:`IterDomainAttr` is optional metadata on 

10 :class:`~qat.experimental.dialect.q1_scf.ir.ops.ForOp` describing the linear 

11 iteration domain of a counted loop. It records the ``start``, ``stop`` and 

12 ``step`` of the varied quantity, the integer ``count`` of iterations, and an 

13 :class:`IterParameter` tag naming the varied quantity and its unit. The attribute 

14 carries no control-flow semantics. 

15 

16Reference: https://docs.qblox.com/en/main/products/qblox_instruments/q1/index.html 

17""" 

18 

19from __future__ import annotations 

20 

21import math 

22from enum import auto 

23 

24from xdsl.dialects.builtin import FloatAttr, IntAttr 

25from xdsl.ir import (

35class IterParameter(StrEnum): 

36 """The quantity varied across iterations, which also fixes the domain unit.""" 

37 

38 frequency = auto() # Hz 

39 phase = auto() # rad 

40 gain = auto() # dimensionless, normalised DAC scale 

41 duration = auto() # ns 

42 

43 

44@irdl_attr_definition 

45class IterParameterAttr(EnumAttribute[IterParameter], SpacedOpaqueSyntaxAttribute): 

46 """Attribute wrapper carrying an :class:`IterParameter`.""" 

47 

48 name = "q1_scf.iter_parameter"

51@irdl_attr_definition 

52class IterDomainAttr(ParametrizedAttribute): 

53 """Linear iteration domain of a :class:`~qat.experimental.dialect.q1_scf.ir.ops.ForOp`. 

54 

55 The domain is the affine sequence ``start, start + step, ...`` truncated to 

56 ``count`` points, all measured in the unit implied by ``parameter``. It records 

57 the physical meaning of a counted loop and does not alter its control-flow 

58 semantics. 

59 

60 :param start: First value of the varied quantity. 

61 :param stop: Exclusive upper bound of the varied quantity. 

62 :param step: Increment between successive iterations; must be non-zero. 

63 :param count: Number of loop iterations, which must equal ``ceil((stop - start) / step)``. 

64 :param parameter: The varied quantity and, implicitly, its unit. 

65 """ 

66 

67 name = "q1_scf.iter_domain" 

68 

69 start: FloatAttr = param_def(FloatAttr) 

70 stop: FloatAttr = param_def(FloatAttr) 

71 step: FloatAttr = param_def(FloatAttr) 

72 count: IntAttr = param_def(IntAttr) 

73 parameter: IterParameterAttr = param_def(IterParameterAttr) 

74

77 start: float, 

78 stop: float, 

79 step: float, 

80 count: int, 

81 parameter: IterParameter, 

82 ): 

83 super().__init__( 

84 FloatAttr(start, 64), 

85 FloatAttr(stop, 64), 

86 FloatAttr(step, 64), 

87 IntAttr(count), 

88 IterParameterAttr(parameter), 

89 )

91 def verify(self) -> None: 

92 """Verify the domain is a well-formed linear iteration. 

93 

94 :raises VerifyException: If ``step`` is zero, if ``count`` is negative, or 

95 if ``count`` does not equal ``ceil((stop - start) / step)``. 

96 """ 

97 start = self.start.value.data 

98 stop = self.stop.value.data 

99 step = self.step.value.data 

100 if step == 0.0: 

101 raise VerifyException("q1_scf.iter_domain step must be non-zero") 

102 

103 if self.count.data < 0: 

104 raise VerifyException( 

105 f"q1_scf.iter_domain count {self.count.data} must be non-negative" 

106 ) 

107 

108 expected = math.ceil((stop - start) / step) 

109 if self.count.data != expected: 

110 raise VerifyException( 

111 f"q1_scf.iter_domain count {self.count.data} does not equal" 

112 f" ceil((stop - start) / step) = {expected}" 

113 )
