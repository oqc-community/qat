14def extract_constant_scalar(ssa: SSAValue) -> float | complex | None: 

15 """Return the Python scalar behind ``ssa`` if it is a compile-time constant. 

16 

17 Handles both pulse-dialect ``ConstantOp`` values (which fold to a 

18 :class:`PulseNumericTypedAttr`) and standard ``arith.constant`` values (which 

19 fold to a :class:`FloatAttr`). Returns ``None`` otherwise. 

20 

21 Complex values whose imaginary part is exactly zero are narrowed to ``float``, 

22 so waveform fields typed strictly as ``float`` accept scalars extracted from an 

23 :class:`AmplitudeAttr`, which always stores its literal value as ``complex``. 

24 """ 

25 

26 attr = ConstantLike.get_constant_value(ssa) 

27 if isinstance(attr, PulseNumericTypedAttr): 

28 value = attr.literal_value 

29 elif isinstance(attr, FloatAttr): 

30 value = attr.value.data 

31 else: 

32 return None 

33 if isinstance(value, complex) and value.imag == 0: 

34 return value.real 

35 return value
